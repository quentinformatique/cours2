// Requête qui recherche et renvoie tous les utilisateurs
MATCH (u:Utilisateur)
RETURN u;

// Requête qui recherche et renvoie tous les enseignants
MATCH (u:Utilisateur:Enseignant)
RETURN u;

// Requête paramétrée qui recherche et renvoie toutes les matières enseignées par un enseignant
MATCH (e:Enseignant {identifiant: $identifiant})-[:Enseigne]->(m:Matiere)
RETURN m;

// Requête paramétrée qui recherche et renvoie tous les étudiants inscrits dans une matière
MATCH (e:Etudiant)-[:Etudie]->(m:Matiere {intitule: $intitule})
RETURN e;

// Requête paramétrée qui affecte une note à un étudiant dans une matière qu’il étudie
MATCH (e:Etudiant {identifiant: $identifiant_etudiant})-[:Etudie]->(m:Matiere {intitule: $intitule_matiere})
SET e.note = $valeur_note;

// Requête paramétrée qui renvoie la note d'un étudiant dans une matière qu'il étudie
MATCH (e:Etudiant {identifiant: $identifiant_etudiant})-[:Etudie]->(m:Matiere {intitule: $intitule_matiere})
RETURN e.note;

// Requête paramétrée qui recherche et renvoie toutes les notes d'un étudiant
MATCH (e:Etudiant {identifiant: $identifiant_etudiant})-[:Etudie]->(m:Matiere)
RETURN m.intitule, e.note;
